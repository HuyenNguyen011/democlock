## Install

```bash
npm install --save demolock
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'demolock'
import 'demolock/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
